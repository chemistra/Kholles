\begin{questions}
    \question Propagation : $s(x,t) = s(x - ct)$ à droite et $s(x,t) = s(x+ct)$ à gauche.
    \question $T' = \dfrac{1}{1+v/c}T$.
    \question $T'' = (1+v/c)T' = \dfrac{1-v/c}{1+v/c}T$.
